import time
import queue
import json
import os
from datetime import datetime

class VideoFrameDaemon:
    """
    视频帧处理守护进程 - 提供低延迟、高吞吐的视频处理能力
    通过事件队列接收请求，异步处理后将结果放入结果队列
    """
    
    def __init__(self):
        self.request_queue = queue.Queue()
        self.result_queue = queue.Queue()
        self.running = False
        self.thread = None
        self.ffmpeg_available = self._check_ffmpeg()
        
    def _check_ffmpeg(self):
        """检查ffmpeg是否可用"""
        try:
            import subprocess
            subprocess.run(['ffmpeg', '-version'], check=True, stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL)
            return True
        except (subprocess.CalledProcessError, FileNotFoundError):
            return False
    
    def start(self):
        """启动守护进程"""
        if not self.ffmpeg_available:
            print("VideoFrameDaemon: ffmpeg不可用，无法启动")
            return False
            
        if self.running:
            return True
            
        self.running = True
        self.thread = threading.Thread(target=self._worker, daemon=True)
        self.thread.start()
        print("VideoFrameDaemon: 已启动")
        return True
    
    def stop(self):
        """停止守护进程"""
        self.running = False
        if self.thread and self.thread.is_alive():
            self.thread.join(timeout=2)
        print("VideoFrameDaemon: 已停止")
    
    def submit_request(self, request_id, video_path, **kwargs):
        """提交视频处理请求"""
        if not self.running:
            return False
            
        request = {
            'request_id': request_id,
            'video_path': video_path,
            'timestamp': time.time(),
            'params': kwargs
        }
        self.request_queue.put(request)
        return True
    
    def get_result(self, timeout=0):
        """获取处理结果"""
        try:
            if timeout > 0:
                return self.result_queue.get(timeout=timeout)
            else:
                return self.result_queue.get_nowait()
        except queue.Empty:
            return None
    
    def _worker(self):
        """工作线程"""
        while self.running:
            try:
                # 等待请求，最多1秒超时以检查running状态
                request = self.request_queue.get(timeout=1)
                
                result = self._process_video(request)
                
                # 将结果放入结果队列
                self.result_queue.put(result)
                self.request_queue.task_done()
                
            except queue.Empty:
                continue
            except Exception as e:
                error_result = {
                    'request_id': request.get('request_id', 'unknown'),
                    'error': str(e),
                    'timestamp': time.time()
                }
                self.result_queue.put(error_result)
                self.request_queue.task_done()
    
    def _process_video(self, request):
        """处理单个视频请求"""
        import subprocess
        import json
        
        video_path = request['video_path']
        params = request.get('params', {})
        request_id = request['request_id']
        
        # 参数提取
        output_dir = params.get('output_dir', f'./frames_{request_id}')
        interval = params.get('interval', 1.0)
        specific_times = params.get('specific_times', [])
        fps = params.get('fps', None)
        format = params.get('format', 'jpg')
        
        # 验证视频文件
        if not os.path.exists(video_path):
            return {
                'request_id': request_id,
                'error': f'视频文件不存在: {video_path}',
                'timestamp': time.time()
            }
        
        # 创建输出目录
        os.makedirs(output_dir, exist_ok=True)
        
        # 获取视频信息
        try:
            result = subprocess.run([
                'ffprobe', '-v', 'quiet', '-print_format', 'json', '-show_format', '-show_streams', video_path
            ], capture_output=True, text=True, timeout=10)
            
            if result.returncode != 0:
                return {
                    'request_id': request_id,
                    'error': '无法读取视频信息',
                    'timestamp': time.time()
                }
                
            video_info = json.loads(result.stdout)
            video_stream = next((stream for stream in video_info['streams'] if stream['codec_type'] == 'video'), None)
            
            if not video_stream:
                return {
                    'request_id': request_id,
                    'error': '视频文件中未找到视频流',
                    'timestamp': time.time()
                }
            
            duration = float(video_info['format'].get('duration', 0))
            
        except Exception as e:
            return {
                'request_id': request_id,
                'error': f'读取视频信息失败: {str(e)}',
                'timestamp': time.time()
            }
        
        extracted_frames = []
        metadata = []
        
        try:
            if specific_times:
                # 按特定时间点抽取
                for i, time_point in enumerate(specific_times):
                    if time_point > duration:
                        continue
                        
                    output_path = os.path.join(output_dir, f'frame_{i:04d}_{time_point:.2f}s.{format}')
                    
                    cmd = [
                        'ffmpeg', '-i', video_path,
                        '-ss', str(time_point),
                        '-vframes', '1',
                        '-y',
                        output_path
                    ]
                    
                    subprocess.run(cmd, stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL, timeout=30)
                    
                    if os.path.exists(output_path):
                        frame_info = {
                            'frame_index': i,
                            'time_point': time_point,
                            'output_path': output_path,
                            'timestamp': datetime.fromtimestamp(time.time()).isoformat()
                        }
                        extracted_frames.append(output_path)
                        metadata.append(frame_info)
                        
            elif fps:
                # 按目标帧率抽取
                output_pattern = os.path.join(output_dir, f'frame_%04d.{format}')
                
                cmd = [
                    'ffmpeg', '-i', video_path,
                    '-vf', f'fps={fps}',
                    '-y',
                    output_pattern
                ]
                
                subprocess.run(cmd, stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL, timeout=60)
                
                # 收集生成的帧
                for file in sorted(os.listdir(output_dir)):
                    if file.endswith(f'.{format}'):
                        frame_path = os.path.join(output_dir, file)
                        extracted_frames.append(frame_path)
                
                # 计算时间戳
                if duration > 0 and extracted_frames:
                    total_frames = len(extracted_frames)
                    frame_interval = duration / total_frames
                    
                    for i, frame_path in enumerate(extracted_frames):
                        time_point = i * frame_interval
                        frame_info = {
                            'frame_index': i,
                            'time_point': time_point,
                            'output_path': frame_path,
                            'timestamp': datetime.fromtimestamp(time.time()).isoformat()
                        }
                        metadata.append(frame_info)
            else:
                # 按时间间隔抽取
                current_time = 0
                frame_count = 0
                
                while current_time <= duration:
                    output_path = os.path.join(output_dir, f'frame_{frame_count:04d}_{current_time:.2f}s.{format}')
                    
                    cmd = [
                        'ffmpeg', '-i', video_path,
                        '-ss', str(current_time),
                        '-vframes', '1',
                        '-y',
                        output_path
                    ]
                    
                    subprocess.run(cmd, stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL, timeout=30)
                    
                    if os.path.exists(output_path):
                        frame_info = {
                            'frame_index': frame_count,
                            'time_point': current_time,
                            'output_path': output_path,
                            'timestamp': datetime.fromtimestamp(time.time()).isoformat()
                        }
                        extracted_frames.append(output_path)
                        metadata.append(frame_info)
                        frame_count += 1
                    
                    current_time += interval
        
        except subprocess.TimeoutExpired:
            return {
                'request_id': request_id,
                'error': '帧抽取超时',
                'timestamp': time.time()
            }
        except Exception as e:
            return {
                'request_id': request_id,
                'error': f'帧抽取失败: {str(e)}',
                'timestamp': time.time()
            }
        
        # 保存元数据
        metadata_path = os.path.join(output_dir, 'frame_metadata.json')
        with open(metadata_path, 'w', encoding='utf-8') as f:
            json.dump({
                'video_source': video_path,
                'extract_method': 'interval' if not specific_times and not fps else ('specific_times' if specific_times else 'fps'),
                'total_frames': len(extracted_frames),
                'extracted_frames': metadata
            }, f, ensure_ascii=False, indent=2)
        
        return {
            'request_id': request_id,
            'success': True,
            'total_extracted': len(extracted_frames),
            'extracted_frames': extracted_frames,
            'metadata_path': metadata_path,
            'video_duration': duration,
            'output_dir': output_dir,
            'timestamp': time.time()
        }

# 全局守护进程实例
video_frame_daemon = VideoFrameDaemon()

def start_video_frame_daemon():
    """启动视频帧守护进程"""
    return video_frame_daemon.start()

def stop_video_frame_daemon():
    """停止视频帧守护进程"""
    video_frame_daemon.stop()

def submit_video_frame_request(request_id, video_path, **kwargs):
    """提交视频帧处理请求"""
    return video_frame_daemon.submit_request(request_id, video_path, **kwargs)

def get_video_frame_result(timeout=0):
    """获取视频帧处理结果"""
    return video_frame_daemon.get_result(timeout)
